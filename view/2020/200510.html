                    wangshanjian 2020-05-10 javaScript plugin

                    Node.js 是一个基于 Chrome V8 引擎的 JavaScript运行环境。 Node.js 使用了一个事件驱动、非阻塞式 I/O 的模型。

                    Node 是一个让 JavaScript 运行在服务端的开发平台，它让 JavaScript 成为与PHP、Python、Perl、Ruby 等服务端语言平起平坐的脚本语言。

                    # mongoose db配置

                    const mongoose = require('mongoose');
const Schema = mongoose.Schema

//1.链接数据库
mongoose.connect('mongodb://localhost/test');

//制定数据模式
const userSchema = new Schema({
  username: {
    type: String,
    required: true
  },
  password: {
    type: String,
    required: true
  },
  email: {
    type: String
  }
})

//3.发布模型
const User = mongoose.model('User', userSchema)

/*
4.添加数据
 */
const admin = new User({
  username: 'zs',
  password: '153',
  email: '123@qq.com'
})

admin.save().then((ret) => {
  console.log('保存成功')
  console.log(ret)
});

/*
5.查询数据
 */
User.findOne({
   username: 'zs'
}).then((res) => console.log(res))

/*
6.删除数据 
 */
User.remove({
   username: 'zs'
 }).then((res) => {
   console.log('删除成功')
   console.log(res)
})

/*
7.更新数据 
 */
User.findByIdAndUpdate('5eb2633a561d3d3f047d5196', {
  password: '123'
}).then((res) => {
  console.log('更新成功')
})

                    # 设置本地API

                    const express = require('express')
const bodyParser = require('body-parser')
const cors = require('cors')
const mysql = require('mysql')

const app = express()

app.use(cors())
app.use(bodyParser.json())
app.use(bodyParser.urlencoded({
  extended: false
}))

const option = {
  host: 'localhost',
  user: 'root',
  password: '123456',
  port: '3306',
  database: 'mytest',
  connectTimeout: 5000,
  multipleStatements: false
}

const conn = mysql.createConnection(option)

app.all('/', (req, res) => {
  conn.query(`SELECT * FROM student `, (e, r) => res.json(new Result({
    data: r
  })))
})

function Result({
  code = 1,
  msg = '',
  data = {}
}) {
  this.code = code
  this.msg = msg
  this.data = data
}


app.listen(5000, () => console.log('running'))
